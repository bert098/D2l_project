	void removeDuplicates() 	
	{
		if(head == null)
		{
			return;
		}
		Node pointer = head; 
		while( pointer != null )
		{
		    Node before = pointer;
		    while(before.getNext()!=null)
		    {
		        if(before.getNext().getItem().equals(pointer.getItem())) 
		        {
		            before.setNext(before.getNext().getNext()); //deletes string if duplicate found 
		            size--;
		        }
		        else
		            before = before.getNext();
		    }
		    pointer = pointer.getNext();
		}
		Node setTail = head; // reset tail to end of list
		while (setTail.getNext() != null)
		{
			setTail = setTail.getNext();
		}
		tail = setTail;
    }
